import random

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    action = None
    if random.random() >= epsilon: # 
        action = actions[random.randint(0,len(actions)-1)] # 
    else: 
        action = sorted(qline.items(), key=lambda q: q[1])[-1][0] #  Q 
    return action
